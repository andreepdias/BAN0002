CREATE TABLE Usuario_Tipos (id serial PRIMARY KEY, nome varchar(50), descricao text);

CREATE TABLE Usuarios (id serial PRIMARY KEY, nome varchar(200), tipo int, FOREIGN KEY (tipo) REFERENCES Usuario_Tipo (id));

CREATE TABLE Pessoas_Desaparecidas (id serial PRIMARY KEY, RG varchar(20), nome varchar(100), ultimo_local varchar(300), inserido_por int, atualizado_por int, tipo int, FOREIGN KEY (inserido_por) REFERENCES Usuarios (id), FOREIGN KEY (atualizado_por) REFERENCES Usuarios (id));

CREATE TABLE Apelidos (id serial PRIMARY KEY, id_pessoa int, apelido varchar(100), FOREIGN KEY (id_pessoa) REFERENCES Pessoas_Desaparecidas (id));

CREATE TABLE Localizacoes (id serial PRIMARY KEY, id_pessoa int, local varchar(300), data_hora timestamp, FOREIGN KEY (id_pessoa) REFERENCES Pessoas_Desaparecidas (id));

CREATE TABLE Denuncias (id serial PRIMARY KEY, id_usuario int, telefone varchar(50), local_ligacao varchar(300), FOREIGN KEY (id_usuario) REFERENCES Usuarios (id));
